##코드형식을 기억하기위한 c++로 2차원 배열 만들기 코드

int main(){
  __int **arr = new int*[5];__          // 더블포인터 생성해서 정수형포인터를 담을 정수형포인터배열 5칸짜리 만들기 (동적할당) 익숙해지자
  for(int i=0; i<5; i++){
    arr[i] = new int[5];
  }
  본문내용 들어갈 부분
  
  for(int i=0; i<5; i++){
    delete[] arr[i];
  }
  delete[] arr;
  __arr = NULL;__                       // c++에서는 arr = nullptr 로 하는게 안전하다. 왜냐하면, NULL은 0과 같은의미라서 상수로 
  return 0;
}
